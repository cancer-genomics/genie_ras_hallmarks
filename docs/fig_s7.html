    pivot_longer(!c(cancer, hugo_symbol),
                 names_to="hgvsp", values_to="y") %>%
    left_join(select(aa_freq, cancer, hugo_symbol, n),
              by=c("cancer",
                   "hugo_symbol")) %>%
    mutate(cancer=cancerLabels(cancer)) %>%
    filter(cancer != "lung") %>%
    distinct() %>%
    mutate(median=qbeta(0.5, 0.5 + y, 0.5 + n-y),

dat <- aa_freq3 %>%
